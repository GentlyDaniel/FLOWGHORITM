if scelta == "Q":
    print("Inserisci il valore del lato:")
    l = float(input())
    while l <= 0:
        print("Il valore del lato deve essere maggiore a 0. Riscrivi il valore:")
        l = float(input())
    a = l ** 2
else:
    if scelta == "T":
        print("Inserisci il valore della base:")
        b = float(input())
        while b <= 0:
            print("Il valore della base deve essere maggiore a 0. Riscrivi il valore:")
            b = float(input())
        print("Ora inserisci il valore dell'altezza:")
        h = float(input())
        while h <= 0:
            print("Il valore dell'altezza deve essere maggiore a 0. Riscrivi il valore:")
            h = float(input())
        a = b * h / 2
    else:
        print("Devi scegliere tra Q, per il quadrato, e T, per il triangolo.")
        a = 0
print("L'area della figura scelta è:")
print(a)
